## Quick Example

```rust
use ultimo::prelude::*;

#[tokio::main]
async fn main() -> ultimo::Result<()> {
    let mut app = Ultimo::new();

    app.get("/users/:id", |ctx: Context| async move {
        let id: u32 = ctx.param("id")?.parse()?;
        ctx.json(json!({
            "id": id,
            "name": "Alice",
            "email": "alice@example.com"
        })).await
    });

    app.listen("127.0.0.1:3000").await
}
```
